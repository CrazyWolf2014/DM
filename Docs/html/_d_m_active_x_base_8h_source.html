237  bool InsideClippingRegion(const RECT& rcClip) const
238  {
239  // true when top/let or bottom/right corners intersect into client area
240  const POINT* pts = (const POINT*)(&m_rcPos);
241  const POINT* ptc = (const POINT*)(&rcClip);
242  return (::PtInRect(&rcClip, pts[0])||::PtInRect(&rcClip, pts[1])
243  ||::PtInRect(&m_rcPos,ptc[0])||::PtInRect(&m_rcPos, ptc[1]));

246  HRESULT DoVerb(long verb)
247  {
248  HRESULT hr = E_FAIL;
249  do
250  {
251  if (0!=(m_dwMiscStatus&(OLEMISC_INVISIBLEATRUNTIME|OLEMISC_NOUIACTIVATE)))
252  {
253  hr = S_FALSE;
254  break;
255  }
256 
257  if (m_spOleObject != NULL)
258  {
259  HWND hWnd = m_pAxHostDelegate->GetAxHostWindow();
260  if (FALSE==::IsWindow(hWnd))
261  {
262  LOG_INFO("[mid]hgy:窗口没初始化时不要初始化activeX,已被坑一次了！！！！！！！！！！！！！！");
263  hr = S_FALSE;
264  break;
265  }
266  hr = m_spOleObject->DoVerb(verb, NULL, static_cast<IOleClientSite*>(this), 0, hWnd, &m_rcPos);
267  if (verb == OLEIVERB_INPLACEACTIVATE &&SUCCEEDED(hr))
268  {
269  m_bInplaceActive = true;

273  hr = E_UNEXPECTED;

277  } while (false);

332  bool ProcessEvent(UINT uMsg, WPARAM wParam, LPARAM lParam, LRESULT* plResult)
333  {
334  LRESULT lResult = -1L;
335  switch( uMsg )
336  {
337  case WM_PAINT:
338  {
339  lResult = ProcessPaint( reinterpret_cast<HDC>(wParam), reinterpret_cast<LPRECT>(lParam));
340  }
341  break;
342  case WM_LBUTTONDOWN:
343  case WM_MBUTTONDOWN:
344  case WM_RBUTTONDOWN:
345  case WM_LBUTTONUP:
346  case WM_MBUTTONUP:
347  case WM_RBUTTONUP:
348  case WM_LBUTTONDBLCLK:
349  case WM_MBUTTONDBLCLK:
350  case WM_RBUTTONDBLCLK:
351  case WM_MOUSEMOVE:
352  case WM_KEYUP:
353  case WM_KEYDOWN:
354  case WM_CAPTURECHANGED:
355  case WM_CANCELMODE:
356  case WM_SETCURSOR:
357  case WM_SETFOCUS: // internal use
358  case WM_KILLFOCUS: // internal use
359  {
360  BOOL bHandled = TRUE;
361  lResult = OnWindowMessage(uMsg, wParam, lParam, bHandled);
362  }
363  break;
364  default: // unsupported!
365  break;
366  }
367  if (plResult)
368  {
369  *plResult = lResult;
370  }
371  return (lResult != -1L);
